 Основное задание №1
N = int(input("Введите размерность массива:")) #Пользователь задает размер массива
A = [0] * N
print("Введите числа массива:") #Вводит числа которые будут входить в этот массив
for i in range(N):
    A[i] = float(input())
Max = 0                # Задаем конт-значение (оно для максимума меньше наименьшего)
In = 0                 # Вводим переменную индексов
for i in range(len(A)-1,-1,-1):# cпятого элемента до -1 элемента, с шагом -1
    if A[i] > Max:     #Перебираем числа, ищем то что больше конт-значение
        Max = A[i]     #Заменяем это число контр-значением
        In = i         #У этого значения выясняем индекс
for i in range(N):     # Снова пребираем числа
    if i > In:         # Ищем тот индекс, который больше индекса нашего наибольшего числа
        A[i] = 0       # И числа под этим индексом заменяем на 0
print(A)

def intersection_list(list1,list2):
    return set(list1).intersection(list2)
N = int(input("Введите размерность первого массива:"))
A = [0] * N
print("Введите числа первого массива:")
for i in range(N):
    A[i] = int(input())
W = int(input("Введите размерность второго массива:"))
B = [0] * W
print("Введите числа второго массива:")
for i in range(W):
    B[i] = int(input())
print(A)
print(B)
print(intersection_list(A,B))

#Дополнительное задание
def summa (a,b):
    c = a+b
    return c
N = int(input("Введите размерность массива:"))
A = [0] * N
print("Введите числа массива:")
for i in range(N):
    A[i] = int(input())
Min = 10**10
for i in range(N):
    if A[i] < Min:
        Min = A[i]
R = int(input("Введите число на которое должен отличаться минимальный элемент:"))
NCh = summa(R,Min)
print("Получается такое число:", NCh)
K = 0
for i in range(N):
    if A[i] == NCh:
        K += 1
print(K)
